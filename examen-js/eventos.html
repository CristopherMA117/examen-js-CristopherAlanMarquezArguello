// Implementación del sistema de eventos const eventSystem = { eventos: {}, myAddEventListener: function(nombreEvento, callback) { if (!this.eventos[nombreEvento]) { this.eventos[nombreEvento] = []; } this.eventos[nombreEvento].push(callback); // Log para demostración this._logEvent(` Event listener registrado para: ${nombreEvento}`); }, myDispatchEvent: function(nombreEvento, datos) { if (!this.eventos[nombreEvento]) { this._logEvent(` No hay listeners para: ${nombreEvento}`); return; } this.eventos[nombreEvento].forEach(callback => { try { callback({ data: datos }); this._logEvent(` Evento disparado: ${nombreEvento}`); } catch (error) { this._logEvent(` Error en callback: ${error.message}`); } }); }, _logEvent: function(mensaje) { const log = document.getElementById('eventLog'); const entry = document.createElement('div'); entry.className = 'log-entry'; entry.textContent = `${new Date().toLocaleTimeString()}: ${mensaje}`; log.appendChild(entry); log.scrollTop = log.scrollHeight; } }; // Funciones globales para usar desde HTML function myAddEventListener(nombreEvento, callback) { eventSystem.myAddEventListener(nombreEvento, callback); } function myDispatchEvent(nombreEvento, datos) { eventSystem.myDispatchEvent(nombreEvento, datos); }
