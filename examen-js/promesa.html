// Función que simula el request al servicio function simularServicio(exitoso) { return new Promise((resolve, reject) => { const tiempoRespuesta = exitoso ? 8000 : 15000; // 8s exitoso, 15s fallido setTimeout(() => { if (exitoso) { resolve('Esta es la respuesta del servicio'); } else { reject(new Error('Error: el servicio no está disponible')); } }, tiempoRespuesta); }); } // Función principal que usa Promise.race() function myMethod(exitoso) { const timeoutPromise = new Promise((_, reject) => { setTimeout(() => { reject(new Error('Error: el servicio no responde en 12 segundos')); }, 12000); }); const servicePromise = simularServicio(exitoso); return Promise.race([servicePromise, timeoutPromise]) .then(resultado => resultado) .catch(error => error.message); }
